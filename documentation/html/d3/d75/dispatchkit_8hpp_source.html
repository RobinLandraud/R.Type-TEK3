295  static int calculate_arity(const std::vector<Proxy_Function> &t_funcs)
296  {
297  if (t_funcs.empty()) {
298  return -1;
299  }
300 
301  const auto arity = t_funcs.front()->get_arity();
302 
303  for (const auto &func : t_funcs)
304  {
305  if (arity != func->get_arity())
306  {
307  // The arities in the list do not match, so it's unspecified
308  return -1;
309  }
310  }
311 
312  return arity;

330  static std::vector<Type_Info> build_type_infos(const std::vector<Proxy_Function> &t_funcs)
331  {
332  auto begin = t_funcs.cbegin();
333  const auto &end = t_funcs.cend();
334 
335  if (begin != end)
336  {
337  std::vector<Type_Info> type_infos = (*begin)->get_param_types();
338 
339  ++begin;
340 
341  bool size_mismatch = false;
342 
343  while (begin != end)
344  {
345  std::vector<Type_Info> param_types = (*begin)->get_param_types();
346 
347  if (param_types.size() != type_infos.size())
348  {
349  size_mismatch = true;
350  }
351 
352  for (size_t i = 0; i < type_infos.size() && i < param_types.size(); ++i)
353  {
354  if (!(type_infos[i] == param_types[i]))
355  {
356  type_infos[i] = detail::Get_Type_Info<Boxed_Value>::get();
357  }
358  }
359 
360  ++begin;
361  }
362 
363  assert(!type_infos.empty() && " type_info vector size is < 0, this is only possible if something else is broken");
364 
365  if (size_mismatch)
366  {
367  type_infos.resize(1);
368  }
369 
370  return type_infos;
371  }
372 
373  return std::vector<Type_Info>();
374  }

657  Boxed_Value get_object(const std::string &name, std::atomic_uint_fast32_t &t_loc, Stack_Holder &t_holder) const
658  {
659  enum class Loc : uint_fast32_t {
660  located = 0x80000000,
661  is_local = 0x40000000,
662  stack_mask = 0x0FFF0000,
663  loc_mask = 0x0000FFFF
664  };
665 
666  uint_fast32_t loc = t_loc;
667 
668  if (loc == 0)
669  {
670  auto &stack = get_stack_data(t_holder);
671 
672  // Is it in the stack?
673  for (auto stack_elem = stack.rbegin(); stack_elem != stack.rend(); ++stack_elem)
674  {
675  for (auto s = stack_elem->begin(); s != stack_elem->end(); ++s )
676  {
677  if (s->first == name) {
678  t_loc = static_cast<uint_fast32_t>(std::distance(stack.rbegin(), stack_elem) << 16)
679  | static_cast<uint_fast32_t>(std::distance(stack_elem->begin(), s))
680  | static_cast<uint_fast32_t>(Loc::located)
681  | static_cast<uint_fast32_t>(Loc::is_local);
682  return s->second;
683  }
684  }
685  }
686 
687  t_loc = static_cast<uint_fast32_t>(Loc::located);
688  } else if ((loc & static_cast<uint_fast32_t>(Loc::is_local)) != 0u) {
689  auto &stack = get_stack_data(t_holder);
690 
691  return stack[stack.size() - 1 - ((loc & static_cast<uint_fast32_t>(Loc::stack_mask)) >> 16)][loc & static_cast<uint_fast32_t>(Loc::loc_mask)].second;
692  }
693 
694  // Is the value we are looking for a global or function?
695  chaiscript::detail::threading::shared_lock<chaiscript::detail::threading::shared_mutex> l(m_mutex);
696 
697  const auto itr = m_state.m_global_objects.find(name);
698  if (itr != m_state.m_global_objects.end())
699  {
700  return itr->second;
701  }
702 
703  // no? is it a function object?
704  auto obj = get_function_object_int(name, loc);
705  if (obj.first != loc) { t_loc = uint_fast32_t(obj.first); }
706 
707  return obj.second;
708 
709  }

967  [this](int l_num_params, const std::vector<Boxed_Value> &l_params, const std::vector<Proxy_Function> &l_funs, const Type_Conversions_State &l_conversions)->Boxed_Value
968  {

970  Boxed_Value bv = dispatch::dispatch(l_funs, attr_params, l_conversions);
971  if (l_num_params < int(l_params.size()) || bv.get_type_info().bare_equal(user_type<dispatch::Proxy_Function_Base>())) {

974  m_e.get().new_scope();
975  m_e.get().add_object("__this", t_bv);
976  }

980  }

983  };

991  } catch (const chaiscript::exception::bad_boxed_cast &) {
992  } catch (const chaiscript::exception::arity_error &) {
993  } catch (const chaiscript::exception::guard_error &) {
994  }
995  throw chaiscript::exception::dispatch_error({l_params.begin() + l_num_params, l_params.end()},
996  std::vector<Const_Proxy_Function>{boxed_cast<Const_Proxy_Function>(bv)});
997  } catch (const chaiscript::exception::bad_boxed_cast &) {
998  // unable to convert bv into a Proxy_Function_Base
999  throw chaiscript::exception::dispatch_error({l_params.begin() + l_num_params, l_params.end()},
1000  std::vector<Const_Proxy_Function>(l_funs.begin(), l_funs.end()));
1001  }
1002  } else {

1004  }
1005  };
1006

1008  return do_attribute_call(1, params, *funs.second, t_conversions);
1009  } else {

1011 
1012  if (!funs.second->empty()) {
1013  try {
1014  return dispatch::dispatch(*funs.second, params, t_conversions);
1015  } catch(chaiscript::exception::dispatch_error&) {
1016  except = std::current_exception();
1017  }
1018  }

1021  // or there was no matching method

1027 
1028  for (const auto &f : *method_missing_funs)
1029  {
1030  if(f->compare_first_type(params[0], t_conversions)) {
1031  fs.push_back(f);
1032  }
1033  }
1034 
1035  return fs;
1036  }();

1041  for (const auto &f : functions) {
1042  if (f->get_arity() != 2) {
1043  return false;
1044  }

1326  static bool function_less_than(const Proxy_Function &lhs, const Proxy_Function &rhs)
1327  {
1328 
1329  auto dynamic_lhs(std::dynamic_pointer_cast<const dispatch::Dynamic_Proxy_Function>(lhs));
1330  auto dynamic_rhs(std::dynamic_pointer_cast<const dispatch::Dynamic_Proxy_Function>(rhs));
1331 
1332  if (dynamic_lhs && dynamic_rhs)
1333  {
1334  if (dynamic_lhs->get_guard())
1335  {
1336  return dynamic_rhs->get_guard() ? false : true;
1337  } else {
1338  return false;
1339  }
1340  }
1341 
1342  if (dynamic_lhs && !dynamic_rhs)
1343  {
1344  return false;
1345  }
1346 
1347  if (!dynamic_lhs && dynamic_rhs)
1348  {
1349  return true;
1350  }
1351 
1352  const auto &lhsparamtypes = lhs->get_param_types();
1353  const auto &rhsparamtypes = rhs->get_param_types();
1354 
1355  const auto lhssize = lhsparamtypes.size();
1356  const auto rhssize = rhsparamtypes.size();
1357 
1358  static const auto boxed_type = user_type<Boxed_Value>();
1359  static const auto boxed_pod_type = user_type<Boxed_Number>();
1360 
1361  for (size_t i = 1; i < lhssize && i < rhssize; ++i)
1362  {
1363  const Type_Info &lt = lhsparamtypes[i];
1364  const Type_Info &rt = rhsparamtypes[i];
1365 
1366  if (lt.bare_equal(rt) && lt.is_const() == rt.is_const())
1367  {
1368  continue; // The first two types are essentially the same, next iteration
1369  }
1370 
1371  // const is after non-const for the same type
1372  if (lt.bare_equal(rt) && lt.is_const() && !rt.is_const())
1373  {
1374  return false;
1375  }
1376 
1377  if (lt.bare_equal(rt) && !lt.is_const())
1378  {
1379  return true;
1380  }
1381 
1382  // boxed_values are sorted last
1383  if (lt.bare_equal(boxed_type))
1384  {
1385  return false;
1386  }
1387 
1388  if (rt.bare_equal(boxed_type))
1389  {
1390  return true;
1391  }
1392 
1393  if (lt.bare_equal(boxed_pod_type))
1394  {
1395  return false;
1396  }
1397 
1398  if (rt.bare_equal(boxed_pod_type))
1399  {
1400  return true;
1401  }
1402 
1403  // otherwise, we want to sort by typeid
1404  return lt < rt;
1405  }
1406 
1407  return false;
1408  }

1413  static void add_keyed_value(Container &t_c, const Key &t_key, Value &&t_value)
1414  {

1416

1418  t_c.reserve(t_c.size() + 1); // tightly control growth of memory usage here
1419  t_c.emplace_back(t_key, std::forward<Value>(t_value));
1420  } else {

1422  *itr = value_type(t_key, std::forward<Value>(t_value));
1423  }
