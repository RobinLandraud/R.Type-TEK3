208  static bool skip_bom(std::ifstream &infile) {
209  size_t bytes_needed = 3;
210  char buffer[3];
211 
212  memset(buffer, '\0', bytes_needed);
213 
214  infile.read(buffer, static_cast<std::streamsize>(bytes_needed));
215 
216  if ((buffer[0] == '\xef')
217  && (buffer[1] == '\xbb')
218  && (buffer[2] == '\xbf')) {
219 
220  infile.seekg(3);
221  return true;
222  }
223 
224  infile.seekg(0);
225 
226  return false;
227  }
228 
230  static std::string load_file(const std::string &t_filename) {
231  std::ifstream infile(t_filename.c_str(), std::ios::in | std::ios::ate | std::ios::binary );
232 
233  if (!infile.is_open()) {
234  throw chaiscript::exception::file_not_found_error(t_filename);
235  }
236 
237  auto size = infile.tellg();
238  infile.seekg(0, std::ios::beg);
239 
240  assert(size >= 0);
241 
242  if (skip_bom(infile)) {
243  size-=3; // decrement the BOM size from file size, otherwise we'll get parsing errors
244  assert(size >= 0); //and check if there's more text
245  }
246 
247  if (size == std::streampos(0))
248  {
249  return std::string();
250  } else {
251  std::vector<char> v(static_cast<size_t>(size));
252  infile.read(&v[0], static_cast<std::streamsize>(size));
253  return std::string(v.begin(), v.end());
254  }
255  }

282 
283  union cast_union
284  {
285  Boxed_Value (ChaiScript_Basic::*in_ptr)(const std::string&);
286  void *out_ptr;
287  };
288 
289  Dl_info rInfo;
290  memset( &rInfo, 0, sizeof(rInfo) );
291  cast_union u;
292  u.in_ptr = &ChaiScript_Basic::use;
293  if ( (dladdr(static_cast<void*>(u.out_ptr), &rInfo) != 0) && (rInfo.dli_fname != nullptr) ) {
294  std::string dllpath(rInfo.dli_fname);
295  const size_t lastslash = dllpath.rfind('/');
296  if (lastslash != std::string::npos)
297  {
298  dllpath.erase(lastslash);
299  }
300 
301  // Let's see if this is a link that we should expand
302  std::vector<char> buf(2048);
303  const auto pathlen = readlink(dllpath.c_str(), &buf.front(), buf.size());
304  if (pathlen > 0 && static_cast<size_t>(pathlen) < buf.size())
305  {
306  dllpath = std::string(&buf.front(), static_cast<size_t>(pathlen));
307  }
308 
309  m_module_paths.insert(m_module_paths.begin(), dllpath+"/");
310  }
311 #endif

393  Boxed_Value use(const std::string &t_filename)
394  {
395  for (const auto &path : m_use_paths)
396  {
397  const auto appendedpath = path + t_filename;
398  try {
399 
400  chaiscript::detail::threading::unique_lock<chaiscript::detail::threading::recursive_mutex> l(m_use_mutex);
401  chaiscript::detail::threading::unique_lock<chaiscript::detail::threading::shared_mutex> l2(m_mutex);
402 
403  Boxed_Value retval;
404 
405  if (m_used_files.count(appendedpath) == 0)
406  {
407  l2.unlock();
408  retval = eval_file(appendedpath);
409  l2.lock();
410  m_used_files.insert(appendedpath);
411  }
412 
413  return retval; // return, we loaded it, or it was already loaded
414  } catch (const exception::file_not_found_error &e) {
415  if (e.filename != appendedpath) {
416  // a nested file include failed
417  throw;
418  }
419  // failed to load, try the next path
420  }
421  }
422 
423  // failed to load by any name
424  throw exception::file_not_found_error(t_filename);
425  }

594  std::string load_module(const std::string &t_module_name)
595  {
596 #ifdef CHAISCRIPT_NO_DYNLOAD
597  (void)t_module_name; // -Wunused-parameter
598  throw chaiscript::exception::load_module_error("Loadable module support was disabled (CHAISCRIPT_NO_DYNLOAD)");
599 #else
600  std::vector<exception::load_module_error> errors;
601  std::string version_stripped_name = t_module_name;
602  size_t version_pos = version_stripped_name.find("-" + Build_Info::version());
603  if (version_pos != std::string::npos)
604  {
605  version_stripped_name.erase(version_pos);
606  }
607 
608  std::vector<std::string> prefixes{"lib", "cyg", ""};
609 
610  std::vector<std::string> postfixes{".dll", ".so", ".bundle", ""};
611 
612  for (auto & elem : m_module_paths)
613  {
614  for (auto & prefix : prefixes)
615  {
616  for (auto & postfix : postfixes)
617  {
618  try {
619  const auto name = elem + prefix + t_module_name + postfix;
620  // std::cerr << "trying location: " << name << '\n';
621  load_module(version_stripped_name, name);
622  return name;
623  } catch (const chaiscript::exception::load_module_error &e) {
624  // std::cerr << "error: " << e.what() << '\n';
625  errors.push_back(e);
626  // Try next set
627  }
628  }
629  }
630  }
631 
632  throw chaiscript::exception::load_module_error(t_module_name, errors);
633 #endif
634  }

644  {

747  void import(const std::string& t_namespace_name)
748  {
749  chaiscript::detail::threading::unique_lock<chaiscript::detail::threading::recursive_mutex> l(m_use_mutex);
750 
751  if (m_engine.get_scripting_objects().count(t_namespace_name)) {
752  throw std::runtime_error("Namespace: " + t_namespace_name + " was already defined");
753  }
754  else if (m_namespace_generators.count(t_namespace_name)) {
